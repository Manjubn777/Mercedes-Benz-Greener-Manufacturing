#based the time taken to close complaints grouping the complaints
def minToCategory(minutes):
    if (math.isnan(minutes)):
        return 'Unspecified'
    elif (minutes < 60.0):
        return 'Fast'
    elif (180.0 > minutes >= 60.0):
        return 'Medium'
    elif (360.0 > minutes >= 180.0):
        return 'Slow'
    else:
        return 'Very Slow'
